inline int bfs()
{
    memset(last, -1, sizeof(last));
    queue<int> q;
    q.push(s);
    flow[s] = INF;
    while (!q.empty())
    {
        int p = q.front();
        q.pop();
        if (p == t) // еҲ°иҫҫжұҮзӮ№пјҢз»“жқҹжҗңзҙў
            break;
        for (int eg = head[p]; eg; eg = edges[eg].next)
        {
            int to = edges[eg].to, vol = edges[eg].w;
            if (vol > 0 && last[to] == -1) // еҰӮжһңж®ӢдҪҷе®№йҮҸеӨ§дәҺ0дё”жңӘи®ҝй—®иҝҮпјҲжүҖд»ҘlastдҝқжҢҒеңЁ-1пјү
            {
                last[to] = eg;
                flow[to] = min(flow[p], vol);
                q.push(to);
            }
        }
    }
    return last[t] != -1;
}
inline int EK()
{
    int maxflow = 0;
    while (bfs())
    {
        maxflow += flow[t];
        for (int i = t; i != s; i = edges[last[i] ^ 1].to) 
        // д»ҺжұҮзӮ№еҺҹи·Ҝиҝ”еӣһжӣҙж–°ж®ӢдҪҷе®№йҮҸ
        {
            edges[last[i]].w -= flow[t];
            edges[last[i] ^ 1].w += flow[t];
        }
    }
    return maxflow;
}

void tarjan(int u) { 
    low[u] = dfn[u] = ++dfncnt, s[++tp] = u, in_stack[u] = 1; 
    for (int i = h[u]; i; i = e[i].nex) { 
        const int &v = e[i].t; 
        if (!dfn[v]) { 
            tarjan(v); 
            low[u] = min(low[u], low[v]); 
        } else if (in_stack[v]) { 
            low[u] = min(low[u], dfn[v]); 
        } 
    } if (dfn[u] == low[u]) { 
        ++sc; 
        while (s[tp] != u) { 
            scc[s[tp]] = sc; 
            sz[sc]++; 
            in_stack[s[tp]] = 0; 
            --tp; 
        } scc[s[tp]] = sc; 
        sz[sc]++; 
        in_stack[s[tp]] = 0; --tp; 
    } 
}
